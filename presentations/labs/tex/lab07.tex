% Task 2
\begin{frame}{Task 2: Study docker-compose.yml}
\begin{center}
\begin{tabular}{|l|l|l|}

% Task 3
\begin{frame}[fragile]{Task 3: Start Services}
\begin{lstlisting}[language=bash]
docker-compose up -d
\end{lstlisting}
\vfill
\begin{center}
\fcolorbox{spublue}{spulight}{\parbox{0.8\textwidth}{
\centering
\textbf{-d} flag runs containers in detached mode\\(background process)
}}
\end{center}
\end{frame}

% Task 4
\begin{frame}[fragile]{Task 4: Check Status}
\begin{lstlisting}[language=bash]
docker-compose ps
\end{lstlisting}
\vfill
\textbf{All services should show:}
\begin{itemize}
    \item State: \texttt{Up}
    \item Ports mapped correctly
\end{itemize}
\end{frame}

% Task 5
\begin{frame}[fragile]{Task 5: View Logs}
\begin{lstlisting}[language=bash]
# View all logs
docker-compose logs -f

# View specific service
docker-compose logs -f backend_container
\end{lstlisting}
\vfill
Press \texttt{Ctrl+C} to exit log view.
\end{frame}

% Task 6
\begin{frame}{Task 6: Test Application}
\begin{center}
Open in browser:\\[1em]
\Large\textcolor{spublue}{\url{http://localhost:8501}}
\end{center}
\vfill
\begin{itemize}
    \item Test the complete RAG application
    \item All services communicate via Docker network
\end{itemize}
\end{frame}

% Task 7
\begin{frame}[fragile]{Task 7: Stop Services}
\begin{lstlisting}[language=bash]
docker-compose down
\end{lstlisting}
\vfill
\textbf{Options:}
\begin{itemize}
    \item \texttt{docker-compose down} - Stop and remove containers
    \item \texttt{docker-compose down -v} - Also remove volumes
\end{itemize}
\end{frame}
