% Task 2
\begin{frame}[fragile]{Task 2: Run OpenSearch}
\begin{lstlisting}[language=bash]
docker run -d --name opensearch-node \
  -p 9200:9200 -p 9600:9600 \
  -e "discovery.type=single-node" \
  -e "DISABLE_SECURITY_PLUGIN=true" \
  opensearchproject/opensearch:2.11.1
\end{lstlisting}
\vfill
\textbf{Ports:}
\begin{itemize}
    \item \texttt{9200} - REST API
    \item \texttt{9600} - Performance Analyzer
\end{itemize}
\end{frame}

% Task 3
\begin{frame}[fragile]{Task 3: Verify OpenSearch}
\begin{lstlisting}[language=bash]
curl http://localhost:9200
\end{lstlisting}
\vfill
\textbf{Expected response:}
\begin{lstlisting}[language=bash]
{
  "name" : "opensearch-node",
  "cluster_name" : "docker-cluster",
  "version" : { "number" : "2.11.1" },
  ...
}
\end{lstlisting}
\end{frame}

% Task 4
\begin{frame}[fragile]{Task 4: Setup Hybrid Search Pipeline}
\begin{lstlisting}[language=bash,basicstyle=\ttfamily\footnotesize]
curl -X PUT "localhost:9200/_search/pipeline/hybrid-search-pipeline" \
  -H "Content-Type: application/json" \
  -d '{
    "phase_results_processors": [{
      "normalization-processor": {
        "normalization": {"technique": "min_max"},
        "combination": {
          "technique": "arithmetic_mean",
          "parameters": {"weights": [0.3, 0.7]}
        }
      }
    }]
  }'
\end{lstlisting}
\end{frame}

% Task 5
\begin{frame}[fragile]{Task 5: Check Health}
\begin{lstlisting}[language=bash]
curl http://localhost:9200/_cluster/health?pretty
\end{lstlisting}
\vfill
\textbf{Look for:}
\begin{itemize}
    \item \texttt{"status": "green"} or \texttt{"yellow"}
    \item \texttt{"number\_of\_nodes": 1}
\end{itemize}
\end{frame}
