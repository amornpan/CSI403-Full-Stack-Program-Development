\section{What is RAG?}

\begin{frame}{The Problem with LLMs}
    \textbf{Large Language Models have limitations:}
    \begin{itemize}
        \item Knowledge cutoff date
        \item No access to your private documents
        \item Can hallucinate (make up facts)
        \item Generic responses
    \end{itemize}
    \vspace{0.5cm}
    \textbf{Solution: RAG (Retrieval-Augmented Generation)}
\end{frame}

\begin{frame}{What is RAG?}
    \begin{center}
        \Large \textbf{Retrieval-Augmented Generation}
    \end{center}
    \vspace{0.5cm}
    \textbf{RAG = Search your documents + Use LLM to generate answers}
    \vspace{0.5cm}
    \begin{enumerate}
        \item \textbf{Retrieval:} Find relevant documents
        \item \textbf{Augmented:} Add context to prompt
        \item \textbf{Generation:} LLM creates answer
    \end{enumerate}
\end{frame}
