\section{FastAPI}

\begin{frame}{Why FastAPI?}
    \begin{itemize}
        \item \textbf{Fast} - High performance
        \item \textbf{Easy} - Intuitive syntax
        \item \textbf{Auto Docs} - Swagger UI
        \item \textbf{Type Hints} - Pydantic validation
        \item \textbf{Async} - Native support
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{FastAPI Example}
    \begin{lstlisting}[language=python]
from fastapi import FastAPI
from pydantic import BaseModel

app = FastAPI(title="RAG API")

class SearchRequest(BaseModel):
    query: str
    top_k: int = 5

@app.get("/health")
def health():
    return {"status": "healthy"}

@app.post("/search")
def search(request: SearchRequest):
    return {"results": []}

\end{frame}
