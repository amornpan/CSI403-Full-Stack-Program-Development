\section{Docker Fundamentals}

\begin{frame}{What is Docker?}
    \textbf{Containerization Platform}
    \begin{itemize}
        \item Package applications with dependencies
        \item Run anywhere (Windows, Mac, Linux)
        \item Isolated environments
        \item Lightweight containers
    \end{itemize}
    \vspace{0.5cm}
    \textbf{Key Concepts:}
    \begin{itemize}
        \item \textbf{Image} - Blueprint/template
        \item \textbf{Container} - Running instance
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Docker Commands}
    \begin{lstlisting}[language=bash]
# List images and containers
docker images
docker ps -a

# Run container
docker run -d --name my-app -p 8080:80 nginx

# Stop and remove
docker stop my-app
docker rm my-app
    \end{lstlisting}
\end{frame}

\section{OpenSearch Setup}

\begin{frame}[fragile]{Run OpenSearch}
    \begin{lstlisting}[language=bash]
docker run -d --name opensearch-node \
  -p 9200:9200 -p 9600:9600 \
  -e "discovery.type=single-node" \
  -e "DISABLE_SECURITY_PLUGIN=true" \
  opensearchproject/opensearch:2.11.1

# Verify
curl http://localhost:9200
    \end{lstlisting}
\end{frame}
